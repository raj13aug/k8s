#Yaml

apiVersion: batch/v1
kind: CronJob
metadata:
  name: backup
spec:
  schedule: "30 15 * * 3"
  jobTemplate:
    spec:
      template:
        spec:
          containers:
          - name: backup
            image: busybox:1.28
            imagePullPolicy: IfNotPresent
            command:
            - echo backup-$(date)
          restartPolicy: OnFailure
